\(
    \begin{aligned}

        & \underline{ \text{Disconnected Space},\ !\text{conn}\ (X,d_X)} :\ 
            \exists V^{1,2}_{\neq \varnothing} \subseteq X,\ V^1 \cap V^2 = \varnothing,\ V^1 \cup V^2 = X
            \ \Leftrightarrow\ \exists S_{\neq \varnothing, X} \subset X,\ S = \text{closed, open} 
            \\
        % Connected Space
        & \underline{ \text{Connected Space, conn}\ (X,d_X)} : \ X \neq \varnothing,\ !\text{conn}
            \hspace{30pt}
            % "Unconnected" empty set
            \bullet\ \underline{ \text{``Unconnected'' Set},\ \varnothing } :\ \varnothing \neq\ \text{conn},\ \text{!conn}
            \\
    \end{aligned}
\)

\vspace{10pt}
\(
    \begin{aligned}
        % Connected Sets R^n
        & \underline{ \begin{gathered}
                \text{Connected Set}\\
                A_{conn} \subseteq X \\
            \end{gathered} }
            :\ 
            \text{conn}\ (A, d|_{A\times A})
            \ \Leftrightarrow\ 
            \begin{aligned}
                \forall \big( \nu^{1,2}_{\neq\varnothing} \subseteq A \big)
                    \big( \nu^1 \cap \nu^2 = \varnothing \big)
                    & \big( \nu^{1,2}_{open} \big) \ \Rightarrow\ \nu^1 \cup \nu^2 \neq A 
                    \\

\subsection{Functions, \(f: X \rightarrow Y ;\ (X, d_X) \rightarrow (Y, d_Y);\ S \subseteq X\)}

\vspace{5pt}
\(
    % Bounded Functions
    \underline{ \text{Bounded, bd}\ f } : \ \ \exists M > 0,\ \forall x \in X,\ \Vert f(x) \Vert_{d_Y} < M 
    \ \Leftrightarrow\ \exists \epsilon > 0,\ \exists y_0 \in Y,\ \forall x \in X,\ f(x) \in B(y_0, \epsilon)
\)\\[5pt]
\(\displaystyle
    % Limit[ing value] of f at x_0
    \underline{ \begin{gathered}
        \text{Limit at}\ x_0 \in \overline{S}\\[-5pt]
        \mss{ \lim_{x \rightarrow x_0} f\mss{(x)} = L \in Y }
    \end{gathered} } 
    :\ \ 
    \begin{aligned}
        & \forall \epsilon > 0,\ \exists \delta > 0,\ \forall \underline{x \in S},\
            \Big[ |x_0 {-} x|_{d_X} {< \delta} \hs\hs \Rightarrow \hs | L {-} f(x) |_{d_Y} {< \epsilon}
            \ \Leftrightarrow\ f( B_{x_0}\mss{(\delta)} ) \subset B_{f(x_0)} \mss{( \epsilon )} \Big]
            \\
        & \nhs\nhs \Leftrightarrow\ \forall( \underline{x^{(n)}\in S} )^\infty,\ x^{(n)} {\rightarrow x_0}
            \ \Rightarrow\ f\mss{(x^{(n)})} \nhs \rightarrow \nhs L
            \ \ \Leftrightarrow\ \forall V_{L} \subseteq Y,\ 
            \exists \underline{ V_{S, x_0} \subseteq S},\ f( V_{x_0} ) \subseteq V_L
    \end{aligned}
\)\\[5pt]
\(\displaystyle
    % Continuous at x_0
    \underline{ \begin{gathered}
        \text{Continuous at}\ x_0 \in X  %\\[-5pt]
        % \mss{ \lim_{x \rightarrow x_0} \hspace{-3pt} {\scriptstyle f(x) = f(x_0) = \text{cont.}\ f|_{x_0} } } \\
    \end{gathered} }
    : \ \text{cont.}\ f|_{x_0} 
    = \lim_{x \in X}^{x_0} f(x) \rightarrow f(x_0) 
    = \lim_{x\rightarrow x_0} f(x) \rightarrow f(x_0) 
\)\\[5pt]
\(
    % Continuous Function
    \begin{aligned}
        \underline{ \text{Continuous, cont}\ f } : \ \ & \forall \epsilon > 0,\ 
            \forall x_0 \in S,\ \exists \delta > 0,\ \forall x \in S,\ 
            |x_0 - x|_{d_X} < \delta \ \Rightarrow\ | f(x_0) - f(x) |_{d_Y} < \epsilon
            \\
        \Leftrightarrow\ \forall x \in S,\ \text{cont}\ f|_x & 
            \ \begin{aligned}
                & \Leftrightarrow\ \forall(x^{(n)})^\infty,\ x^{(n)} \rightarrow x_0 
                    \ \Rightarrow\ f\mss{(x^{(n)})} \rightarrow f\mss{(x_0)}
                    & \Leftrightarrow 
                    & \ \ \forall V^1 \subseteq Y,\ f^{-1}(V^1) = V^2 \subseteq X
                    \\
                &  \Leftrightarrow\ \forall(x^{(n)})^\infty,\ x^{(n)} \rightarrow 0 
                    \ \Rightarrow\ f{\scriptstyle(x-x^{(n)})} \rightarrow_p f\mss{(x)}
                    & \Leftrightarrow 
                    & \ \ \forall U^1 \subseteq Y,\ f^{-1}(U^1) = U^2 \subseteq X
                    \\
            \end{aligned}
    \end{aligned}
\)\\[5pt]
\(
    % Results of Continuous Functions
    \begin{aligned}
        \hspace{3pt} & \bullet\ \forall K_1 \subseteq X,\ f(K_1) = K_2 \subseteq Y \\
        & \bullet\ \forall S^1_{conn} \subseteq X,\ f(S^1) = S^2_{conn} \subseteq Y\\
        & \bullet\ \text{\scriptsize Space cont,bd \(f\) is complete subspace of bd \(f\)}
    \end{aligned}
    \hfill
    \begin{aligned}
        & \bullet\ (K, d_K),\ (Y, d_Y),\ \big( \text{cont.}\ f \Leftrightarrow \text{u.c.}\ f \big)\\
        & \bullet\ (K,d_K),\ f: K \rightarrow \mathbb{R} \ \Rightarrow\ \text{bd}\ f\\
        & \bullet\ (K_{\neq \varnothing},d_K),\ f: K \rightarrow \mathbb{R},\ \Rightarrow\ \exists (p_1,p_2) 
            \mss{ \big( \hs \forall p,\ F(p_1) \leq F(p) \leq F(p_2) \hs \big) }
            \\
    \end{aligned}
    \hfill
\)\\[10pt]
\(
    % Uniformly Continuous Function
    \begin{aligned}
        \underline{ \text{Uniformly Cont., u.c}\ f } : \ \ & 
            \forall \epsilon > 0,\ \exists \delta > 0,\ \forall x_0 \in X,\ \forall x \in X,\
            |x_0 - x|_{d_X} < \delta \ \Rightarrow\ | f(x_0) - f(x) |_{d_Y} < \epsilon
            \\
        & \Leftrightarrow\ \forall(x^{(n)})^\infty,\ x^{(n)} \rightarrow 0 
            \ \Rightarrow\ f{\scriptstyle(x-x^{(n)})} \rightarrow_u f\mss{(x)}
            \\
        & \bullet\ \text{u.c}\ f \ \Rightarrow\ \text{cont.}\ f 
            \hspace{15pt} \bullet\ (K, d_K),\ (Y, d_Y),\ \big( \text{cont.}\ f \Leftrightarrow \text{u.c.}\ f \big)
            \\
    \end{aligned}
\)
