\section{Analytic/Holomorphic Functions}

\vspace{-10pt}
% Derivative
\(\begin{aligned}
	& \underline{\text{Differentiable}}:\ \exists f'\mss{(z)} = \lim_{h \rightarrow 0} \tfrac{f\mss{(z + h)} - f\mss{(z)}}{z} 
		\hspace{20pt} \mss{ \begin{aligned}
			& f = u + iv\\[-1pt]
			& h = \sigma + i \tau
		\end{aligned} }
		\hspace{5pt} \ \Rightarrow\ \mss{ 
			\begin{gathered}
				\underline{\text{Cauchy-Riemann Eq.}}\\
				\boxed{ \partial f / \partial \bar{z} = 0 }
			\end{gathered}:\
			\boxed{ \begin{aligned}
				\frac{\partial u}{\partial x} = \frac{\partial v}{\partial y} \ & 
					,\ \ \frac{\partial v}{\partial x} = - \frac{\partial u}{\partial y}
					\\[2pt]
				r \frac{\partial u}{\partial r} = \frac{\partial v}{\partial \theta} \ & 
					,\ \ r \frac{\partial v}{\partial r} = - \frac{\partial u}{\partial \theta}
			\end{aligned} }
		}
		\\[5pt]
	& \bullet\ \Big( \text{C-R Eq.}\big|_z \Big) , \Big( f\mss{(z)} \in C^1 \Big) \ \Rightarrow\ \exists f'\mss{(z)}
		\hspace{20pt} \bullet\ \tfrac{\partial}{\partial z} 
		= \tfrac{\partial x}{\partial z} \tfrac{\partial}{\partial x} + \tfrac{\partial y}{\partial z} \tfrac{\partial}{\partial y}
		\hspace{5pt} , \ \mss{ \begin{aligned}
			& x = \tfrac{1}{2}(z + \bar{z})\\[-1pt]
			& y = \tfrac{1}{2i}(z - \bar{z})
		\end{aligned} }
		\\[5pt]
	& \bullet\ f'\mss{(z)} = \tfrac{\partial u}{\partial x} + i \tfrac{\partial v}{\partial x}
		\hspace{20pt} \bullet\ \Delta u = \tfrac{\partial^2 u}{\partial^2 x} + \tfrac{\partial^2 u}{\partial^2 y} = 0
		\hspace{20pt} \bullet\ u\mss{(z)} = c \text{ \ OR \ } u^2 + v^2 = c \ \Rightarrow\ f\mss{(z)} = c
\end{aligned}\)

\vspace{10pt}
% Definitions
\(\begin{aligned}
	& \underline{\text{Holomorphic}}:\ \forall z \in D ,\ \exists f'\mss{(z)}
		\hspace{20pt} \underline{\text{Smooth}}:\ f\mss{(z)} \in C^\infty
		\hspace{20pt} \underline{\text{Entire}}:\ \text{Analytic everywhere}
		\\[10pt]
	& \underline{\text{Analytic}}:\ f\mss{(z_0)} \in C^\omega \subset C^\infty:\ 
		\exists \delta>0,\ \forall |z| < \delta,\ f\mss{(z_0+z)} = \mss{\sum} a_n z^n
		\ \rightarrow\ \boxed{ f\mss{(z)} = \mss{\sum}\hs a_n \mss{(z-z_0)}^n }
		\\[5pt]
	& \bullet\ \mss{\sum}\hs a_n \mss{(z_1-z_0)}^n 
		\ \Rightarrow\ \mss{\sum}\hs \big| a_n \mss{(z-z_0)}^n \big| :\ \boxed{ |z-z_0| < |z_1-z_0| }
		\hspace{20pt} \bullet\ \boxed{ a_n = \tfrac{f^{n} (z_0)}{n!} }
		\\[5pt]
	% Convergence Tests
	& \bullet\ \underline{\text{Root Test}}:\ \lim \tfrac{|a_{n+1}|}{|a_n|} = \tfrac{1}{R}
		\hspace{20pt} \bullet\ \underline{\text{Ratio Test}}:\ \lim \sqrt[n]{a_n} = \tfrac{1}{R}
		\hspace{20pt} \bullet\ \tfrac{1}{R} = \limsup \sqrt[n]{a_n}
\end{aligned}\)

\(
	\begin{aligned}
		% Sum
		& \underline{\text{Sum}}:\hs
			\mss{ \int_{\gamma_1 + \gamma_2} } \hspace{-15pt} f\mss{(z)} \hs dz 
			= \mss{ \int_{\gamma_1} } f\mss{(z)} \hs dz + \mss{ \int_{\gamma_2} } f\mss{(z)} \hs dz 
			\\[5pt]
		% Green's Theorem
		& \begin{gathered}
				\underline{\text{Green's}}\\[-3pt]
				\underline{\text{Theorem}}
			\end{gathered}:\hs 
			\mss{\oint}\hs u\hs dx + v\hs dy 
			= \mss{\iint} \big[ \tfrac{\partial v}{\partial x} - \tfrac{\partial u}{\partial y} \big] dx\hs dy
	\end{aligned}
	\hfill
	\vrule
	\hfill
	% Flux/Curl
	\arraycolsep=2pt\begin{array}{c c l c l}
		\displaystyle \mss{\oint}\hs\hs \overline{f\mss{(z)}} \hs\hs dz & =
			& \displaystyle \mss{\int} \big[ u \hs dx + v \hs dy \big]
			& + 
			& \displaystyle i \mss{\int} \big[ u \hs dy - v \hs dx \big]
			\\[8pt]
		\underline{ \tss{CR for \(\bar{f} \ \rightarrow\)} } & =
			& \displaystyle \mss{\iint} \big[ \tfrac{\partial v}{\partial x} - \tfrac{\partial u}{\partial y} \big] dy \hs dx
			& + 
			& \displaystyle i \mss{\iint} \big[ \tfrac{\partial u}{\partial x} + \tfrac{\partial v}{\partial y} \big] dy \hs dx
			\\[8pt]
		\begin{gathered}
				\underline{\tss{sourceless}}\\[-7pt]
				\underline{\tss{irrotational}}
			\end{gathered} & = 
			& \displaystyle \mss{\oint}\hs\hs \hsvec{f} \cdot d\hsvec{l} \hspace{8pt} \text{\scriptsize(curl)}
			& + 
			& \displaystyle i \mss{\oint}\hs\hs \big( \hsvec{f} \cdot \hat{n} \big) \hs dl \hspace{8pt} \text{\scriptsize(flux)}
	\end{array}
\)

% Cauchy's Theorem
\(\begin{aligned}
	& \underline{\text{Cauchy's Theorem}}:\ \boxed{ \oint_\gamma f\mss{(z)} \hs dz = 0 }
		= i \iint \frac{\partial f}{\partial x} + i \frac{\partial f}{\partial y} \ dx\hs dy
		\hspace{20pt} \Big( \hs \begin{gathered}
			\text{\scriptsize \(f'\) must be cont. to use Green's Theorem}\\[-5pt]
			\text{\scriptsize Goursat proves w/o cont. w/ triangles}
		\end{gathered} \hs \Big)
		\\
	% Proof
	& 1.\ \text{\scriptsize simple closed} 
		\hspace{15pt} 2.\ \text{\scriptsize closed squares/triangles+\(\Box\)}
		\hspace{15pt} 3.\ \exists F \ (F' = f, \text{\scriptsize well-defined, path ind.})
		\hspace{15pt} 4.\ \Box\ \big( \mss{\oint f dz = \oint F' dz = 0} \big)
		\\[3pt]
	% Antiderivative
	& \bullet\ D \text{ is simp.-con.} \ \Rightarrow\ \exists F \ (\text{\scriptsize cont.}\ F' = f, \ \text{holo.})
		% Logarithms
		\hspace{20pt} \bullet\ \text{\scriptsize no zero} 
		\hs \Rightarrow\hs \boxed{f\mss{(z)} = e^{g(z)}} 
		\ , \hs \underline{ \mss{g(z) = \text{Log}\ f(z_0) + \int_{z_0}^z \frac{f'}{f} \hs dw} }
	% Morera's Theorem
		\\
	& \bullet\ \text{Morera's Theorem}:\ 
		f \text{ is cont.},\ \forall \gamma_{\in C^1} \in D ,\ \oint_\gamma f\mss{(z)}\hs dz = 0 
		\ \Rightarrow\ f \text{ is holo. in } D
\end{aligned}\)

\vspace{10pt}
% Cauchy's Formula
\(\begin{aligned}
	& \underline{\text{Cauchy's Formula}}:\ \boxed{ f\mss{(z_0)} = \frac{1}{2\pi i} \oint \frac{f\mss{(z)}}{z-z_0} \hs dz }
		\ \Rightarrow\ f\mss{(z)} = \mss{\sum}\hs a_n (z-z_0)^n \hspace{15pt} \text{\scriptsize(Power series/analytic)}
		\\[5pt]
	& 1.\ f\mss{(z_0)} = \lim_{r \rightarrow 0}\ \tfrac{1}{2\pi i} 
		\mss{ \int_0^{2\pi} } \tfrac{f (z_0 + r e^{i\theta})}{r e^{i\theta}} \hs i r e^{i\theta} \hs d\theta
		= \begin{gathered}[t]
			\boxed{ \tfrac{1}{2\pi} \mss{ \int_0^{2\pi} } f \mss{(z_0 + r e^{i\theta})} \hs d\theta }\\[-2pt]
			\text{\scriptsize(Mean Value Theorem)}
		\end{gathered}
		\leq \max \big| f \mss{(z_0 + r e^{i\theta})} \big|
		\ \rightarrow\ 0 \hspace{10pt} \text{\scriptsize(if \(f\) is cont.)}
		\\[5pt]
	% Louisville's Theorem
	& \bullet\ \text{Louisville's Theorem}:\ 
		f \text{ is entire},\ \exists M>0,\ f\mss{(z)} \leq M \ \Rightarrow\ f\mss{(z)} = c
		\\[5pt]
	& \bullet\ \text{Analytic}:\ \exists F (\text{\scriptsize holo.},\ \text{\scriptsize cont.}\ F') 
		\ \Rightarrow\ F = \mss{\sum} b_n\mss{(z-z_0)}^n 
		\ \Rightarrow\ \text{\scriptsize cont.}\ f = \mss{\sum} a_n\mss{(z-z_0)}^n 
		\ \Rightarrow\ \underline{ \text{\scriptsize cont.}\ f' }
		\\
	& \bullet\ \boxed{ a_n = \tfrac{1}{2\pi i} \mss{\oint} \tfrac{ f\mss{(z)} }{ (z - z_0)^{k+1} } \hs dz }
		= \tfrac{f^{n}(z_0)}{n!}
		\hspace{15pt} \bullet\ \exists z_0, \ \forall k,\ f^{(k)}\mss{(z_0)} = 0 \ \Rightarrow\ f\mss{(z)} = 0
\end{aligned}\)

						\text{Removable}\\[-5pt]

					\end{gathered} }:\ \ & f\mss{(z)} = \mss{\sum_0}\hs a_n \mss{(z-z_0)}^n 
					\hspace{10pt} \mss{(m = 0)}
					\ =\ a_0 + \dots
					\\
				\text{Pole}:\ \ & f\mss{(z)} = \mss{\sum_{-m}}\hs a_n \mss{(z-z_0)}^n 
					\hspace{10pt} \mss{(m \geq 1)}
					\ =\ \tfrac{1}{g(z)} = \tfrac{H(z) = \frac{1}{h(z)}}{(z-z_0)^m}
					\\
				\mss{ \begin{gathered}
					\text{Essential}\\[-5pt]
					\text{Singularity}
				\end{gathered} }:\ \ & f\mss{(z)} = \mss{\sum_{-\infty}}\hs a_n \mss{(z-z_0)}^n 
					\hspace{10pt} \mss{(m = \infty)}
			\end{aligned}
	\end{aligned}
	\hfill
	% Residue
	\begin{aligned}[t]
		& \underline{\text{Residue}}:\ \text{Res}\hs \mss{(f;z_0)}

			\\[5pt]
		& \bullet\ \boxed{ \mss{\oint_\gamma} \hs f\mss{(\zeta)} \hs\hs d\zeta 
			= 2\pi i\ \mss{\sum_{\text{sing.}}}\ \text{Res}\hs \mss{(f;z_0)} }
			\\
		& \bullet\ g\mss{(z)} = \mss{(z-z_0)}^j 
			\ \Rightarrow\ \mss{
				\text{Res}\hs (g;z_0) = \left\{ \begin{aligned}
					0 & \hspace{10pt} j \neq -1\\
					1 & \hspace{10pt} j = -1\\

			}
			\\[3pt]

		& \ast\ f\mss{(z)} = \tfrac{H(z)}{z-z_0} \rightarrow \text{Res}\hs \mss{(f;z_0)} = H\mss{(z_0)}\\[3pt]
		& \bullet\ G'\mss{(z_0)} \neq 0 \rightarrow \text{Res}\hs \big( \mss{\frac{H}{G};z_0} \big) = \tfrac{H(z_0)}{G'(z_0)}
	\end{aligned}

		= \mss{\sum_{0}^\infty} \hs a_n \mss{(z-z_0)}^n

\(\begin{aligned}
	% Green's\Stokes' Theorem
	\begin{gathered}
		\text{\underline{Green's/Stokes'}}\\[-3pt]
		\text{\underline{Theorem}}
	\end{gathered}: & &
		{\oint} 
		\left[ \begin{matrix}
			u \\ v
		\end{matrix} \right]
		\cdot 
		\left[ \begin{matrix}
			dx \\ dy
		\end{matrix} \right]
		& \ =\hs {\iint} \left| \arraycolsep=1pt \begin{matrix}
			\nabla_x & \nabla_y\\[2pt]
			\mss{-} ( \mss{-} u ) & v
		\end{matrix} \right| 
		\hs dx \hs dy
		\\[3pt]
	& &
		{\oint} \left[ \begin{matrix}
			v \\ -u
		\end{matrix} \right]
		\cdot 
		\left[ \begin{matrix}
			dy \\ -dx
		\end{matrix} \right]
		& \ =\hs {\iint} \hsvec{\nabla}
		\cdot 
		\left[ \begin{matrix}
			v \\ -u
		\end{matrix} \right]
		\hs dx \hs dy
		\\[5pt]
	% 2D Divergence
	\text{\underline{2D Div. Theorem}}: & &
		{\oint} \left[ \begin{matrix}
			v \\ -u
		\end{matrix} \right]
		\cdot 
		\hat{n} \ dl
		& \ =\hs {\iint} \hsvec{\nabla}
		\cdot 
		\left[ \begin{matrix}
			v \\ -u
		\end{matrix} \right]
		\hs dA
\end{aligned}\)

% Green's 1st Identity
\(
	\begin{aligned}
		\begin{gathered}
			\text{\underline{Green's 2D}}\\[-3pt]
			\text{\underline{1st Identity}}
		\end{gathered}: & & 
			{\oint} \ \mss{
				\left[ \begin{matrix}
					f \nabla_x g \\[3pt]
					f \nabla_y g 
				\end{matrix} \right]
			} 
			\cdot 
			\hat{n} \ dl
			& \hs\hs =\hs {\iint} \hsvec{\nabla} \cdot \big[ f \hsvec{\nabla} g \big] \hs dA
			\\
		& & \Aboxed{
				{\oint} \ \big[ f \hsvec{\nabla} g \big] 
				\cdot 
				\hat{n} \ dl 
				& \hs\hs = \hs {\iint} \left[ f \hsvec{\nabla}^2 g + \hsvec{\nabla} g \cdot \hsvec{\nabla} f \right] dA 
			}
	\end{aligned}
	\hspace{-6pt} \Rightarrow \hs \mss{ \displaystyle \underline{ 
		\oint \big[ f \hsvec{\nabla} f \big] \cdot \hat{n} \ dl 
		= \hs \iint \left[ f \hsvec{\nabla}^2 f + \Vert \hsvec{\nabla} f \Vert^2 \right] dA 
	} } 
\)

% Green's 2nd Identity
\(\begin{aligned}
	\begin{gathered}
		\text{\underline{Green's 2D}}\\[-3pt]
		\text{\underline{2nd Identity}}
	\end{gathered}: & & \Aboxed{ 
			{\oint} \ \big[ f \hsvec{\nabla} g - g \hsvec{\nabla} f \big] 
			\cdot 
			\hat{n} \ dl 
			& \ =\hs {\iint} \left[ f \hsvec{\nabla}^2 g - g \hsvec{\nabla}^2 f \right] dA 
		}
\end{aligned}\)

\vspace{15pt}
% Green's 3rd Identity
\(\begin{aligned}
	\begin{gathered}
		\text{\underline{Green's 2D}}\\[-3pt]
		\text{\underline{3rd Identity}}
	\end{gathered}: & & 
		\boxed{ \mss{ \hsvec{\nabla}^2 G = \delta^2 {(z-z_0)} } }
		\ \Rightarrow\ \Aboxed{ 
			f \mss{(z_0)} 
			& \ = \ {\oint} \ \big[ f \hsvec{\nabla} G - {G} \hsvec{\nabla} f \big] \cdot \hat{n} \ dl 
			+ {\iint} \left[ G { \hsvec{\nabla}^2 f } \right] dA 
		}
		\\
	& & \Aboxed{ 
			f \mss{(z_0)} & \ = \ {\oint} \ f\mss{(z)} \big[ \hsvec{\nabla} G \cdot \hat{n} \big] dz 
			\hspace{15pt}
			\mss{ \begin{aligned}
				& \bullet\ f \text{ is harmonic}\\[-1pt]
				& \bullet\ G \text{ is } 0 \text{ on the boundary}
			\end{aligned} }
		}
\end{aligned}\)
