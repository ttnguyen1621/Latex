\begin{eqnarray*}
	k_n &=& \frac{2\pi n}{\lambda} \ \ \Rightarrow \ \ \Delta k = \frac{2\pi}{\lambda} \\ \\
	\Psi(x) &?\approx&\ \displaystyle\sum \limits_{n \ = \ -\infty}^{\infty} \frac{1}{\lambda} \left[ \int_{-\lambda/2}^{\lambda/2} \Psi(x) \ e^{-i \frac{2\pi n}{\lambda} x} \ dx \right] e^{i \frac{2 \pi n}{\lambda} x} \\ \\
	&=&\ \displaystyle\sum \limits_{k_n \ = \ -\infty}^{\infty} \frac{\Delta k}{\sqrt{2\pi}} \left[ \frac{1}{\sqrt{2\pi}} \int_{-\lambda/2}^{\lambda/2} \Psi(x) \ e^{-i k_n x} \ dx \right] e^{i k_n x} \\ \\ \\ \\
	\lim_{\lambda \rightarrow \infty} \ \ \ \Psi(x) &=& \frac{1}{\sqrt{2\pi}} \int_{-\infty}^{\infty} \ \left[ \frac{1}{\sqrt{2\pi}} \int_{-\infty}^{\infty} \Psi(x) \ e^{-i k x} \ dx \right] e^{i k x} \ dk \\ \\
	&=& \frac{1}{\sqrt{2\pi}} \int_{-\infty}^{\infty} \ \hat{\Psi}(k) e^{i k x} \ dk \\ \\
	\hat{\Psi}(k) &=& \frac{1}{\sqrt{2\pi}} \int_{-\infty}^{\infty} \Psi(x) e^{-i k x} \ dx \\ \\ \\
	&or& \\ \\ \\
	x(t) &=& \int_{-\infty}^{\infty} \ \hat{x}(f) e^{i 2\pi f t} \ df \\ \\
	\hat{x}(f) &=& \int_{-\infty}^{\infty} \ x(t) e^{-i 2\pi f t} \ dt \\ \\
\end{eqnarray*}
Proof:
\begin{eqnarray*}
	\int_{-\infty}^{\infty} \ \hat{x}(f) (e^{-\epsilon f^2}) \ e^{i 2\pi f t} \ df &=& a(t) \\ \\
	&=& \int_{-\infty}^{\infty} \ \left[ \int_{-\infty}^{\infty} x(\tau) e^{-i 2\pi f \tau} \ d\tau \right] (e^{-\epsilon f^2}) \ e^{i 2\pi f t} \ df \\ \\
	&=& \int_{-\infty}^{\infty} \ x(\tau) \left[ \int_{-\infty}^{\infty} (e^{-\epsilon f^2}) \ e^{- i 2\pi f (\tau-t)} \ df \right] \ d\tau \\ \\
	&=& \int_{-\infty}^{\infty} \ x(\tau'+t) \left[ \int_{-\infty}^{\infty} (e^{-\epsilon f^2}) \ e^{- i 2\pi f\tau'} \ df \right] \ d\tau' \\ \\
	&=& \int_{-\infty}^{\infty} \ x(\tau'+t) \ \left( \frac{1}{2\sqrt{\pi}\epsilon} \ e^{\frac{-(\tau')^2}{4\epsilon^2}} \right) \ d\tau' \\ \\
	&=& \int_{-\infty}^{\infty} \ x(\epsilon\tau''+t) \ \left( \frac{1}{\sqrt{2\pi}\sqrt{2}} e^{ \frac{-(\tau'')^2}{2\sqrt{2}^2} } \right) \ d\tau'' \\ \\ \\
	\int_{-\infty}^{\infty} \ \hat{x}(f) \ e^{i 2\pi f t} &=& \lim_{\epsilon\rightarrow 0} \ \int_{-\infty}^{\infty} \ \hat{x}(t) (e^{-\epsilon f^2}) \ e^{i 2\pi f t} \ df \ = \ \\ \\
	\lim_{\epsilon\rightarrow 0} \ a(t) &=& x(t) \int_{-\infty}^{\infty} \frac{1}{\sqrt{2\pi}\sigma} e^{ \frac{-(\tau'')^2}{2\sigma^2} } \ d\tau'' \\ \\
	&=& x(t) \\
\end{eqnarray*}

% Z Transform (Discrete Laplace)
\section{Z Transform (Discrete Laplace)}

\(\displaystyle
	\hspace{20pt}
	\begin{aligned}
		& \int_0^\infty \frac{ u\mss{(x)} }{ e^{sx} } \hs dx\\[5pt]
		& \bullet\ \mss{ |u(x)| \leq Me^{ax} }
	\end{aligned}
	\ \ \ \begin{gathered}
		\Rightarrow\\[-5pt]
		{\scriptstyle(x_n = n \Delta_x)}
	\end{gathered}
	\ \ \ \ \begin{aligned}
		& \sum_{n=0}^\infty \frac{ u\mss{(x_n)} \Delta_x }{ [e^{s\Delta_x}]^{n} } 
			= \sum_{n=0}^\infty \frac{a_n}{z^n}
			\\[5pt]
		& \bullet\ \mss{ |a_n| \leq MR^n }
	\end{aligned}
\)

\boldmath \begin{gather*}
	(Z\{ a_n \})(z) = \sum_{n \ = \ 0}^{\infty} \ \frac{a_n}{z^n} 
		\ \ \mss{ 
			\leq \sum_{n = 0}^\infty \frac{MR^n}{z^n}
			\ \leftarrow\ \underline{|z| > R} 
		}
\end{gather*} \unboldmath

% Convolution Property
\(
	{\textbf{Convolution}}:\ \boxed{ f_n = a_n \boldsymbol{*} y_n = \sum\limits_{k = 0}^{n} a_k y_{n-k} }
	\ \Rightarrow\ \mss{ \begin{aligned}
		& Z\{ f_n \} = Z\{ a_n \} Z\{ y_n \} \\
		& Z\{ y_n \} = \tfrac{1}{ Z\{ a_n \} } Z \{ f_n \} = Z\{g_n\} Z\{f_n\}
	\end{aligned} }
	\ \Rightarrow\ \underline{y_n = g_n \boldsymbol{*} f_n}
\)

% Shifting Property
\(
	{\textbf{Shifting}}:\ \underline{ Z\{ a_{n+m} \} = a_m + \frac{a_{m+1}}{z} + \frac{a_{m+2}}{z^2} + ... } 
	= \boxed{ \left[ Z\{ a_n \} - \mss{ \sum\limits_{j=0}^{m-1} } \hs \frac{a_j}{z^j} \right] z^m }
\)

\vspace{15pt}
% Example
\(
	\begin{gathered}
		\bullet\ f_{n+2} = y_{n+2} - 3 y_{n+1} + 2y_n \\[-2pt]
		\mss{( a = \{1, -3, 2, 0, 0, \dots\} )} 
	\end{gathered}
	\ \rightarrow \
	\begin{gathered}
		{ Z\{f_n\} = Z\{y_n\} \left[ 1 + \tfrac{-3}{z} + \tfrac{2}{z^2} \right] }\\[-4pt]
		\mss{( Z\{a_n\} = \text{\underline{System Transfer Func.}} )}
	\end{gathered}
	\ \rightarrow \
	\mss{ \begin{aligned}
		Z\{y_n\} & = \left[ \tfrac{2}{1-(2/z)} + \tfrac{-1}{1-(1/z)} \right] Z\{f_n\} \\
		& =  Z\{ \mss{ \underline{ g_n = 2^{n+1} - 1} } \} Z\{f_n\}
	\end{aligned} }
\)

\(
	\ast\ \begin{aligned}[t]

			= Z\{g_n\} \left[ \tfrac{f_0}{z^0} + \tfrac{f_1}{z^1} + \tfrac{0}{z^2} + ... \right]

	\end{aligned} 
	\hfill \vline \hfill
	\begin{aligned}[t]

			\\[5pt]
		& \tfrac{y_0}{z^0} + \tfrac{y_1}{z^1} + \tfrac{y_2}{z^2} + ... 
			= Z\{g_n\} \left[ \tfrac{f_0}{z^0} + \tfrac{f_1}{z^1} + \tfrac{f_2}{z^2} + ... \right]
