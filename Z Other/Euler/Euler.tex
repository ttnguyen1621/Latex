\( \displaystyle
	\lim_{n\rightarrow\infty} (1 + \tfrac{x}{n})^n 
	= \lim_{n\rightarrow\infty} \mss{ \sum_{i=0}^n } \mss{ \binom{n}{i} } \left( \tfrac{x}{n} \right)^i
	= \lim_{n\rightarrow\infty} \mss{ \sum_{i=0}^n } \tfrac{n!/(n-i)!}{n^i} \tfrac{x^i}{i!}
	= \mss{ \sum_{i=0}^\infty } \tfrac{x^i}{i!}
	= f(x):
	\mss{ \begin{aligned}
		f(0) & = 1\\
		f(1) & \equiv e
	\end{aligned} }
\)

\vspace{10pt}
\(
	% x < y -> f(x) < f(y) 
	\bullet\ x < y \ \Rightarrow\ \underline{ 0 < f(x) < f(y) < \infty }
	\hspace{20pt}
	% f(x) = f(y) -> x = y
	\bullet\ f(x) = f(y) \ \Rightarrow\ x = y
\)

\vspace{10pt}
% f(x+y) = f(x) f(y)
\(
	\displaystyle
	\bullet\ f\mss{(x+y)}
	= \mss{ \sum_{n=0}^\infty } \tfrac{(x+y)^n}{n!} 
	= \mss{ \sum_{n=0}^\infty } \tfrac{1}{n!} \left[ \mss{ \sum_{i=0}^n } \tfrac{n!}{i!(n-i)!} x^i y^{n-i} \right]
	= \mss{ \sum_{n=0}^\infty } \mss{ \sum_{i=0}^n } \tfrac{x^i}{i!} \tfrac{y^{n-i}}{(n-i)!} 
	= \mss{ \sum_{i=0}^\infty } \tfrac{x^i}{i!} \left[ \mss{ \sum_{j=0}^\infty } \tfrac{y^j}{j!} \right]
	= 
	f\mss{(x)} f\mss{(y)}
\)

\( \displaystyle
	% (e^x)^k works for positive integers, k
	\begin{aligned}[t]
		\mss{ \lim_{n\rightarrow\infty} } (1 + \tfrac{x}{n})^{kn} &
			= \mss{ \lim_{n\rightarrow\infty} } \mss{ \sum_{i=0}^n }
			\left\{ \tfrac{(kn)!}{(kn-i)!} \right\} \tfrac{1}{i!} \left( \tfrac{x}{n} \right)^i
			\\ 
		& = \mss{ \lim_{n\rightarrow\infty} } \mss{ \sum_{i=0}^n } 
			\left\{ \tfrac{(kn)!}{(kn-i)!} \tfrac{1}{(kn)^i} \right\} \tfrac{(kx)^i}{i!}
	\end{aligned}
	\hspace{-5pt}
	= f(kx) = ( f\mss{(x)} )^k
	: \hspace{-5pt}
	\mss{ \begin{aligned}[t]
		\\[-18pt]
		f(0) & = f(0)^k = 1^k\\
		f(k) & = f(1)^k = e^k\\
		f(k)^n & = e^{kn} = (e^k)^n
	\end{aligned} }
	\hspace{-5pt} \vline\hfill
	% let f(x) = e^x
	\begin{gathered}[t]
		\boxed{ e^x \equiv f(x) = \mss{ \sum_{n=0}^\infty } \tfrac{x^n}{n!} }\\[2pt]
		\mss{ \bullet\  e^{x+y} = e^x e^y }\\[-3pt]
		\mss{ \bullet\ \underline{ e^{xk} = (e^x)^k } } 
	\end{gathered}
\)

\(
	\begin{aligned}
		e^{x' \tfrac{d}{dx}} f(x) & =
			\begin{gathered}[b]
				\mss{ \big( dx' = \tfrac{x'}{n} \big) }\\
				\lim_{n\rightarrow\infty} \big( 1 + dx' \tfrac{d}{dx} \big)^n f(x)
			\end{gathered}
			\ \rightarrow\
			e^{dx' \tfrac{d}{dx}} = e^{dx' g(x)}
			\ \rightarrow\ 
			\boxed{ 
				\tfrac{d}{dx} \mss{ \hs \equiv \hs h(x) }
				\ \Rightarrow\
				e^{x' \tfrac{d}{dx}} = e^{ {\displaystyle \mss{ \int_x^{x + x'} } } \hspace{-13pt} g(t) \hs dt} 
			}
			\\
		f(x + x') & = f(x) 
			+ \mss{ \int_x^{x+x'} } \nhs f'\mss{(x_1)} \hs\hs dx_1
			= f(x) 
			+ \mss{ \int_x^{x+x'} } 
			\left[ f'\mss{(x)} + \mss{ \int_x^{x+x_1} } f''\mss{(x_2)} \hs\hs dx_2 \right] dx_1 
			\\
		& = f(x) 
			+ \mss{ \int_x^{x+x'} } 
			\nhs f'\mss{(x)} \hs\hs dx_1 
			+ \mss{ \int_x^{x+x'} \hspace{-5pt} \int_x^{x+x_1} } 
			\left[ f''\mss{(x)} + \mss{ \int_x^{x+x_2} } f^{(3)}\mss{(x_3)} \hs\hs dx_3 \right] \hs\hs dx_2 \hs dx_1
			\\
		& = f(x) 
			+ \mss{ \int_x^{x+x'} } \nhs f'\mss{(x)} \hs\hs dx_1 
			+ \mss{ \int_x^{x+x'} \hspace{-5pt} \int_x^{x+x_1} }
